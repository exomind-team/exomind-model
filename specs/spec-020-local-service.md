    if (options?.model) {
      formData.append('model', options.model);

      method: 'POST',
      body: formData,
    });

    if (!response.ok) {
      throw new Error(`ASR failed: ${response.statusText}`);
    }

    return response.json();
  }

  async synthesize(
    text: string,
    options?: { voiceId?: number; model?: string }
  ): Promise<Blob> {
    const response = await fetch(`${this.baseUrl}/v1/tts/synthesize`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        text,
        voice_id: options?.voiceId || 77,
        model: options?.model || 'vits-zh-hf-fanchen-C',
      }),
    });

    if (!response.ok) {
      throw new Error(`TTS failed: ${response.statusText}`);
    }

    return response.blob();
  }

  async *streamStatus(): AsyncGenerator<StatusUpdate> {
    const ws = new WebSocket(`${this.baseUrl}/v1/ws/status`);

    for await (const message of this.readMessages(ws)) {
      if (message.type === 'status_update') {
        yield message.data;
      }
    }
  }
}
